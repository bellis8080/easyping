#### 1.2 Database Queries

- [x] 1.2.1 Total pings in date range:
  ```sql
  SELECT COUNT(*) FROM pings
  WHERE tenant_id = $1 AND created_at >= $2 AND created_at < $3
    AND status != 'draft';
  ```
- [x] 1.2.2 Average resolution time (minutes):
  ```sql
  SELECT AVG(EXTRACT(EPOCH FROM (resolved_at - created_at)) / 60)
  FROM pings
  WHERE tenant_id = $1 AND resolved_at IS NOT NULL
    AND created_at >= $2 AND created_at < $3;
  ```
- [x] 1.2.3 SLA compliance rate:
  ```sql
  SELECT
    COUNT(*) FILTER (WHERE sla_breached = false AND resolved_at IS NOT NULL) * 100.0 /
    NULLIF(COUNT(*) FILTER (WHERE resolved_at IS NOT NULL), 0)
  FROM pings
  WHERE tenant_id = $1 AND created_at >= $2 AND created_at < $3;
  ```
- [x] 1.2.4 Open pings count:
  ```sql
  SELECT COUNT(*) FROM pings
  WHERE tenant_id = $1 AND status IN ('new', 'in_progress', 'waiting_on_user');
  ```
- [x] 1.2.5 Daily volume:
  ```sql
  SELECT DATE(created_at) as date, COUNT(*) as count
  FROM pings
  WHERE tenant_id = $1 AND created_at >= $2 AND created_at < $3
    AND status != 'draft'
  GROUP BY DATE(created_at)
  ORDER BY date;
  ```
- [x] 1.2.6 Category breakdown:
  ```sql
  SELECT c.id, c.name, COUNT(p.id) as count
  FROM pings p
  LEFT JOIN categories c ON p.category_id = c.id
  WHERE p.tenant_id = $1 AND p.created_at >= $2 AND p.created_at < $3
    AND p.status != 'draft'
  GROUP BY c.id, c.name
  ORDER BY count DESC;
  ```

**KBAnalyticsTab Pattern (copy this for PingAnalyticsTab):**

```typescript
// From components/analytics/KBAnalyticsTab.tsx - lines 143-208
const [data, setData] = useState<KBAnalyticsResponse | null>(null);
const [isLoading, setIsLoading] = useState(true);
const [error, setError] = useState<string | null>(null);

const getDateParams = useCallback(() => {
  const end = new Date();
  const start = new Date();
  switch (dateRange) {
    case '7d': start.setDate(end.getDate() - 7); break;
    case '30d': start.setDate(end.getDate() - 30); break;
    case '90d': start.setDate(end.getDate() - 90); break;
    default: start.setDate(end.getDate() - 7);
  }
  return {
    startDate: start.toISOString().split('T')[0],
    endDate: end.toISOString().split('T')[0],
    period: dateRange,
  };
}, [dateRange]);

useEffect(() => {
  const fetchData = async () => {
    setIsLoading(true);
    setError(null);
    try {
      const params = getDateParams();
      const searchParams = new URLSearchParams({ ... });
      const response = await fetch(`/api/analytics/pings?${searchParams}`);
      // ... handle response
    } catch (err) {
      setError(err instanceof Error ? err.message : 'An error occurred');
    } finally {
      setIsLoading(false);
    }
  };
  fetchData();
}, [getDateParams]);
```

### Time Formatting Helper

```typescript
function formatResolutionTime(minutes: number | null): string {
  if (minutes === null) return 'N/A';
  const hours = Math.floor(minutes / 60);
  const mins = Math.round(minutes % 60);
  if (hours === 0) return `${mins}m`;
  if (mins === 0) return `${hours}h`;
  return `${hours}h ${mins}m`;
}
```
